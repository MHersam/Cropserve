package de.uni_stuttgart.informatik.sopra.sopraapp;

import com.google.android.gms.maps.model.LatLng;

/**
 * Class that capsules attributes for a field w/o a contract
 *
 * @author Patrick Koss
 * @author Sarah Gutierrez Mu√±oz
 * @author Michael Hersam
 */
public class FeldOhneVertrag {
    private int id;
    private String art;
    private double flaeche;
    private LatLng koordinaten[];
    private Region region;
    private Benutzer benutzer;

    /**
     * constructor
     *
     * @param id field w/o contract specific ID
     * @param art field w/o contract type
     * @param flaeche field w/o contract area
     * @param koordinaten field w/o contract coordinates
     * @param region field w/o contract location
     */
    public FeldOhneVertrag(int id, String art, double flaeche, LatLng[] koordinaten, Region region, Benutzer benutzer) {
        this.id = id;
        this.art = art;
        this.flaeche = flaeche;
        this.koordinaten = koordinaten;
        this.region = region;
        this.benutzer = benutzer;
    }

    /**
     * getter Benutzer
     * @return
     */
    public Benutzer getBenutzer() {
        return benutzer;
    }

    /**
     * getter ID
     *
     * @return
     */
    public int getId() {
        return id;
    }

    /**
     * getter Art
     *
     * @return
     */
    public String getArt() {
        return art;
    }

    /**
     * getter Flaeche
     *
     * @return
     */
    public double getFlaeche() {
        return flaeche;
    }

    /**
     * getter Koordinaten
     *
     * @return
     */
    public LatLng[] getKoordinaten() {
        return koordinaten;
    }

    /**
     * getter Region
     *
     * @return
     */
    public Region getRegion() {
        return region;
    }


    /**
     * method edit the Feld
     *
     * @param koordinaten
     * @param region
     */
    public void bearbeiten(LatLng koordinaten[], Region region) {
        this.koordinaten = koordinaten;
        this.region = region;
    }

}
